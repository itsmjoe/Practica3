pipeline 
{
	agent any
	tools {nodejs "node"}
	
	stages 
	{
		stage('Git') 
		{
			steps{ git branch: 'main', url:'https://github.com/itsmjoe/Practica3' }
		}	
	
		stage('Install')
		{
			steps{ bat 'npm install --force' } 
		}
		
		stage('Test')
		{
			steps{ bat 'npm run test:ci' }
		}
		
		stage('Build')
		{
			steps{ bat 'npm run ng build --configuration="production" --optimization --build-optimizer' }
		}

		stage('Deploy')
		{
			steps{ bat 'robocopy dist/frontend C:/Program Files/nginx-1.20.2 /E & EXIT /B 0'}
		}
	
	}
}
